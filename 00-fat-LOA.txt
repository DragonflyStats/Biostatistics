




data( fat )
fat <- data.frame( item=factor(fat$Id),
meth=fat$Obs,
repl=factor(fat$Rep),
y=fat$Sub )

fat.roy = lme(y ~ meth-1, data = fat,
  random = list(item=pdSymm(~ meth-1)), 
  weights=varIdent(form=~1|meth),
  correlation = corSymm(form=~1 | item/repl), 
  method="ML")

tidy(fat.roy) # conditional Residuals

getLOA(fat.roy)
